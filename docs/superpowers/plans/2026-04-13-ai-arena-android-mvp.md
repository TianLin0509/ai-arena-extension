```typescript
// Marker protocol: AI must emit these markers at start/end of each response.
// Round number prevents cross-round contamination when reusing the same chat.

export function startMarker(round: number): string {
  return `ARENA_START_R${round}`;
}

export function doneMarker(round: number): string {
  return `ARENA_DONE_R${round}`;
}

export function buildMarkerInstruction(round: number): string {
  return `\n（请在回答的最开头输出 ${startMarker(round)}，最末尾输出 ${doneMarker(round)} 作为标记，不要解释这些标记）`;
}

export function stripMarkers(text: string): string {
  return text
    .replace(/ARENA_START_R\d+/g, '')
    .replace(/ARENA_DONE_R\d+/g, '')
    .trim();
}
```

```typescript
import { buildMarkerInstruction } from './MarkerProtocol';

export type DebateStyle = 'free' | 'collab';

export interface ResponseEntry {
  name: string;
  text: string;
}

export interface DebateRound {
  roundNum: number;
  style: DebateStyle;
  guidance?: string;
  responses: Record<string, ResponseEntry>;
}

export const DEBATE_STYLES: Record<DebateStyle, { name: string; prompt: string }> = {
  free: {
    name: '自由辩论',
    prompt: '以下是其他 AI 对同一问题的回答，请分析他们的观点，指出你认同和不认同的地方，并给出你的改进回答。',
  },
  collab: {
    name: '群策群力',
    prompt:
      '以下是你的队友们对同一问题的回答。你们是协作关系，目标是共同得出最优方案。请：1) 吸收队友回答中的亮点和你没想到的角度；2) 补充你认为队友遗漏的重要内容；3) 整合所有人的优势，给出一个更完善的综合回答。不要攻击或否定队友，而是取长补短。',
  },
};

export const DebateEngine = {
  buildDebatePrompt(
    participantId: string,
    responses: Record<string, ResponseEntry>,
    style: DebateStyle,
    roundNum: number,
    guidance: string,
    concise: boolean,
  ): string {
    const styleConfig = DEBATE_STYLES[style] || DEBATE_STYLES.free;
    const isCollab = style === 'collab';

    const roundHints: Record<number, string> = isCollab
      ? {
          1: '这是第1轮协作。请仔细阅读队友们的回答，找出各自的亮点和你没想到的角度。',
          2: '这是第2轮协作。队友们已经互相补充了一轮，请在此基础上进一步整合，查漏补缺。',
          3: '这是第3轮协作。方案已趋于成熟，请做最终打磨——精简冗余，强化核心结论，形成一份完整方案。',
        }
      : {
          1: '这是第1轮辩论。请仔细阅读其他参与者的初始回答，找出核心分歧和共识。',
          2: '这是第2轮辩论。经过上一轮交锋，请聚焦于仍存在分歧的关键点，深化你的论证或修正你的观点。',
          3: '这是第3轮辩论。辩论已进入深水区，请避免重复已达成共识的内容，集中攻克剩余分歧点，给出最终立场。',
        };

    const defaultHint = isCollab
      ? `这是第${roundNum}轮协作。请只补充新的见解，不要重复已有内容。`
      : `这是第${roundNum}轮辩论。请只针对仍有分歧的核心问题发表精炼观点。`;
    const roundHint = roundHints[roundNum] || defaultHint;

    const conciseRule = concise
      ? '\n\n⚠️ 简洁模式：请控制回答在 1000 字以内，用要点列表呈现核心观点，避免长篇大论。每个论点简明扼要。'
      : '';

    const othersText = Object.entries(responses)
      .filter(([id, r]) => id !== participantId && r.text)
      .map(([, r]) => `【${r.name} 的回答】:\n${r.text}`)
      .join('\n\n');

    let prompt = `${roundHint}\n\n${styleConfig.prompt}\n\n${othersText}${conciseRule}`;
    if (guidance) prompt = `用户补充要求：${guidance}\n\n${prompt}`;
    return prompt + buildMarkerInstruction(roundNum);
  },

  buildSummaryPrompt(
    originalQuestion: string,
    rounds: DebateRound[],
    responses: Record<string, ResponseEntry>,
    customInstruction: string,
  ): string {
    let historySection = '';
    if (rounds.length > 0) {
      historySection = '\n\n## 辩论历史摘要\n';
      for (const round of rounds) {
        historySection += `\n### 第${round.roundNum}轮（${DEBATE_STYLES[round.style]?.name || round.style}）\n`;
        if (round.guidance) historySection += `用户引导：${round.guidance}\n`;
      }
      historySection += '\n（以上为辩论过程，以下为各方最终观点）\n';
    }

    const allText = Object.values(responses)
      .filter((r) => r.text)
      .map((r) => `【${r.name} 的观点】:\n${r.text}`)
      .join('\n\n');

    let prompt = `你是一场多 AI 辩论的最终裁判。${originalQuestion ? `原始问题是：「${originalQuestion}」\n` : ''}以下是各 AI 的讨论记录（经过 ${rounds.length} 轮辩论）。
${historySection}
${allText}

请你作为裁判，给出结构化的最终总结：

## 共识结论
各方一致认同的核心观点

## 分歧焦点
仍存在争议的地方，列出各方立场

## 最终裁定
综合各方观点后，你认为最准确、最完整的结论是什么

## 实操建议
基于以上讨论，给出可落地的建议

## 标注规则
请对每个结论标注共识度：
- 🟢 全员共识：所有参与者都明确支持此观点
- 🟡 多数认同：多数参与者支持，少数持保留意见
- 🔴 存在争议：参与者之间有明确分歧，列出各方立场
- 💡 独家洞察：仅一方提出但有价值的独特视角

要求：客观公正，不偏袒任何一方，重点是综合各家之长得出最优答案。`;

    if (customInstruction?.trim()) prompt += `\n\n## 额外要求\n${customInstruction.trim()}`;
    return prompt + buildMarkerInstruction(rounds.length + 1);
  },
};
```

```javascript
// src/inject/base.js
// Runs inside every AI platform WebView.
// Per-platform scripts are prepended before this (define SITE + platformOverrides globals).
// This file replaces extension's content-*.js shared logic using postMessage instead of chrome.runtime.

(function() {
  'use strict';

  // ── platform-injected values: SITE, platformOverrides (optional) ──
  var site = typeof SITE === 'string' ? SITE : 'unknown';
  var overrides = (typeof platformOverrides === 'object' && platformOverrides) || {};

  // ── RN bridge ──
  function postToRN(payload) {
    try {
      if (window.ReactNativeWebView && window.ReactNativeWebView.postMessage) {
        window.ReactNativeWebView.postMessage(JSON.stringify(payload));
      }
    } catch (e) {}
  }

  function sleep(ms) { return new Promise(function(r) { setTimeout(r, ms); }); }

  function stripMarkers(text) {
    return String(text || '')
      .replace(/ARENA_START_R\d+/g, '')
      .replace(/ARENA_DONE_R\d+/g, '')
      .trim();
  }

  // ── selectors (injected by platform script as SELECTORS global) ──
  var selectors = (typeof SELECTORS === 'object' && SELECTORS) || {};
  var _selectorCache = {};
  var _failCount = {};
  var _reportedFailures = {};

  function queryBySelectors(action, opts) {
    opts = opts || {};
    var cached = _selectorCache[action];
    if (cached) {
      var el = opts.all ? document.querySelectorAll(cached) : document.querySelector(cached);
      if (opts.all ? el.length > 0 : el) { _failCount[action] = 0; return el; }
      _failCount[action] = (_failCount[action] || 0) + 1;
      if (_failCount[action] >= 3) { delete _selectorCache[action]; _failCount[action] = 0; }
    }
    var arr = selectors[action] || [];
    for (var i = 0; i < arr.length; i++) {
      var sel = arr[i];
      var found = opts.all ? document.querySelectorAll(sel) : document.querySelector(sel);
      if (opts.all ? found.length > 0 : found) {
        _selectorCache[action] = sel;
        return found;
      }
    }
    var heuristic = heuristicQuery(action, opts);
    if (heuristic) return heuristic;
    if (!_reportedFailures[action]) {
      _reportedFailures[action] = true;
      postToRN({ type: 'selectorFailure', site: site, action: action });
    }
    return opts.all ? [] : null;
  }

  function heuristicQuery(action, opts) {
    if (action === 'input') {
      var editables = [].slice.call(document.querySelectorAll('[contenteditable="true"], textarea'));
      if (editables.length) {
        return editables.reduce(function(best, el) {
          var r = el.getBoundingClientRect(), b = best.getBoundingClientRect();
          return r.width * r.height > b.width * b.height ? el : best;
        });
      }
      return null;
    }
    if (action === 'response') {
      var blocks = document.querySelectorAll('div, article, section');
      for (var i = blocks.length - 1; i >= 0; i--) {
        var t = (blocks[i].innerText || '').trim();
        if (t.length > 100 && blocks[i].getBoundingClientRect().height > 50) {
          return opts.all ? [blocks[i]] : blocks[i];
        }
      }
      return opts.all ? [] : null;
    }
    if (action === 'sendButton') {
      var btns = [].slice.call(document.querySelectorAll('button'));
      return btns.filter(function(b) {
        return b.getBoundingClientRect().bottom > window.innerHeight - 150 && b.querySelector('svg');
      }).pop() || null;
    }
    return opts.all ? [] : null;
  }

  // ── response readers ──
  function getLastResponseText() {
    if (overrides.getLastResponseText) return overrides.getLastResponseText(queryBySelectors) || '';
    var els = queryBySelectors('response', { all: true });
    if (els.length) return els[els.length - 1].textContent || '';
    return '';
  }

  function getLastAssistantText() {
    if (overrides.getLastAssistantText) return overrides.getLastAssistantText(queryBySelectors) || '';
    var els = queryBySelectors('response', { all: true });
    if (els.length) return (els[els.length - 1].innerText || '').trim();
    return '';
  }

  // ── injection ──
  async function robustInject(el, text) {
    if (overrides.robustInject) return overrides.robustInject(el, text);
    el.focus();
    if (el.tagName === 'TEXTAREA') {
      var setter = Object.getOwnPropertyDescriptor(window.HTMLTextAreaElement.prototype, 'value');
      if (setter && setter.set) setter.set.call(el, text);
      else el.value = text;
      el.dispatchEvent(new Event('input', { bubbles: true }));
      el.dispatchEvent(new Event('change', { bubbles: true }));
      return;
    }
    el.innerHTML = '';
    el.dispatchEvent(new Event('input', { bubbles: true }));
    await sleep(100);
    try {
      var dt = new DataTransfer();
      dt.setData('text/plain', text);
      el.dispatchEvent(new ClipboardEvent('paste', { clipboardData: dt, bubbles: true, cancelable: true }));
      await sleep(200);
      if ((el.innerText || '').trim().length > 0) return;
    } catch (e) {}
    try {
      el.focus();
      document.execCommand('selectAll', false, null);
      document.execCommand('insertText', false, text);
      await sleep(200);
      if ((el.innerText || '').trim().length > 0) return;
    } catch (e) {}
    el.innerHTML = text.split('\n').map(function(l) { return '<p>' + (l || '<br>') + '</p>'; }).join('');
    el.dispatchEvent(new Event('input', { bubbles: true }));
  }

  async function injectAndSend(text) {
    var el = queryBySelectors('input');
    if (!el) return { status: 'error', error: 'input not found' };
    await robustInject(el, text);
    for (var attempt = 0; attempt < 8; attempt++) {
      await sleep(400);
      var btn = overrides.findSendButton ? overrides.findSendButton(queryBySelectors) : queryBySelectors('sendButton');
      if (btn && !btn.disabled) { btn.click(); return { status: 'sent' }; }
    }
    el.dispatchEvent(new KeyboardEvent('keydown', { key: 'Enter', code: 'Enter', keyCode: 13, which: 13, bubbles: true }));
    el.dispatchEvent(new KeyboardEvent('keyup', { key: 'Enter', code: 'Enter', keyCode: 13, which: 13, bubbles: true }));
    return { status: 'sent' };
  }

  // ── marker watcher ──
  var _observer = null;

  function setupMarkerWatcher(doneMarker) {
    stopMarkerWatcher();
    _observer = new MutationObserver(function() {
      var text = getLastResponseText();
      var tail = text.replace(/\s+$/, '').slice(-200);
      if (tail.indexOf(doneMarker) !== -1) {
        postToRN({
          type: 'markerDetected',
          site: site,
          doneMarker: doneMarker,
          textLength: text.length,
          text: stripMarkers(getLastAssistantText()),
        });
        stopMarkerWatcher();
      } else {
        postToRN({ type: 'streaming', site: site, textLength: text.length });
      }
    });
    _observer.observe(document.body, { childList: true, characterData: true, subtree: true });
  }

  function stopMarkerWatcher() {
    if (_observer) { _observer.disconnect(); _observer = null; }
  }

  // ── RN-callable API ──
  window.__arena_injectAndSend = async function(text) {
    try {
      var r = await injectAndSend(text);
      postToRN({ type: 'injectDone', site: site, ok: r.status === 'sent', error: r.error || null });
    } catch (e) {
      postToRN({ type: 'injectDone', site: site, ok: false, error: String(e && e.message || e) });
    }
  };

  window.__arena_readLatestResponse = function() {
    var text = getLastAssistantText();
    postToRN({ type: 'responseText', site: site, text: stripMarkers(text) });
  };

  window.__arena_setupMarkerWatcher = function(doneMarker) {
    setupMarkerWatcher(doneMarker);
  };

  window.__arena_stopMarkerWatcher = function() { stopMarkerWatcher(); };

  // ── ready signal ──
  postToRN({ type: 'ready', site: site });

  true;  // required by injectedJavaScript convention
})();
```

```typescript
import { useArenaStore } from '../store/arenaStore';

interface WebViewMessage {
  type: 'ready' | 'injectDone' | 'streaming' | 'markerDetected' | 'responseText' | 'selectorFailure';
  site: string;
  ok?: boolean;
  error?: string | null;
  textLength?: number;
  text?: string;
  doneMarker?: string;
  action?: string;
}

export function handleWebViewMessage(participantId: string, rawData: string): void {
  let msg: WebViewMessage;
  try {
    msg = JSON.parse(rawData);
  } catch {
    return; // ignore malformed
  }

  const store = useArenaStore.getState();

  switch (msg.type) {
    case 'ready':
      // informational; no state change required for MVP
      return;
    case 'injectDone':
      if (msg.ok) store.setPollStatus(participantId, 'waiting', 0);
      return;
    case 'streaming':
      store.setPollStatus(participantId, 'streaming', msg.textLength ?? 0);
      return;
    case 'markerDetected':
      store.setPollStatus(participantId, 'ready', msg.textLength ?? 0);
      if (typeof msg.text === 'string') store.setParticipantResponse(participantId, msg.text);
      return;
    case 'responseText':
      if (typeof msg.text === 'string') store.setParticipantResponse(participantId, msg.text);
      return;
    case 'selectorFailure':
      // dev telemetry; could surface later
      if (__DEV__) console.warn('[arena] selector failed:', msg.site, msg.action);
      return;
  }
}
```

```typescript
import { handleWebViewMessage } from '../engine/MessageBus';

export interface ParticipantWebViewHandle {
  injectAndSend: (text: string) => void;
  readLatest: () => void;
  setupMarkerWatch: (doneMarker: string) => void;
  reload: () => void;
}

interface Props {
  participantId: string;
  service: ServiceId;
}

const DESKTOP_UA =
  'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36';

export const ParticipantWebView = forwardRef<ParticipantWebViewHandle, Props>(
  ({ participantId, service }, ref) => {
    const webRef = useRef<WebView>(null);
    const injected = React.useMemo(() => buildInjectedScript(service), [service]);

    useImperativeHandle(ref, () => ({
      injectAndSend: (text: string) => {
        const escaped = JSON.stringify(text);
        webRef.current?.injectJavaScript(`window.__arena_injectAndSend(${escaped}); true;`);
      },
      readLatest: () => {
        webRef.current?.injectJavaScript(`window.__arena_readLatestResponse && window.__arena_readLatestResponse(); true;`);
      },
      setupMarkerWatch: (doneMarker: string) => {
        const escaped = JSON.stringify(doneMarker);
        webRef.current?.injectJavaScript(`window.__arena_setupMarkerWatcher(${escaped}); true;`);
      },
      reload: () => webRef.current?.reload(),
    }));

    const onMessage = (e: WebViewMessageEvent) => {
      handleWebViewMessage(participantId, e.nativeEvent.data);
    };

    return (
      <WebView
        ref={webRef}
        source={{ uri: SERVICES[service].url }}
        userAgent={DESKTOP_UA}
        injectedJavaScript={injected}
        onMessage={onMessage}
        sharedCookiesEnabled
        thirdPartyCookiesEnabled
        javaScriptEnabled
        domStorageEnabled
        originWhitelist={['*']}
        mixedContentMode="always"
        style={{ flex: 1 }}
      />
    );
  },
);
ParticipantWebView.displayName = 'ParticipantWebView';
```

```typescript
import React, { useEffect, useRef, useState } from 'react';
import { SafeAreaView, StatusBar, StyleSheet, View } from 'react-native';
import { useArenaStore } from './src/store/arenaStore';
import { DEFAULT_PARTICIPANTS } from './src/config/services';
import { buildMarkerInstruction, doneMarker } from './src/engine/MarkerProtocol';
import { DebateEngine, ResponseEntry } from './src/engine/DebateEngine';
import { DashboardScreen } from './src/screens/DashboardScreen';
import { DetailScreen } from './src/screens/DetailScreen';
import { ParticipantPool, ParticipantPoolHandle } from './src/components/ParticipantPool';

export default function App() {
  const poolRef = useRef<ParticipantPoolHandle>(null);
  const [detailParticipantId, setDetailParticipantId] = useState<string | null>(null);

  const participants = useArenaStore((s) => s.participants);
  const addParticipant = useArenaStore((s) => s.addParticipant);
  const startBroadcast = useArenaStore((s) => s.startBroadcast);
  const appendDebateRound = useArenaStore((s) => s.appendDebateRound);
  const resetSession = useArenaStore((s) => s.resetSession);
  const markerRound = useArenaStore((s) => s.markerRound);
  const setFlowState = useArenaStore((s) => s.setFlowState);

  // Initialize default participants once on mount.
  useEffect(() => {
    if (useArenaStore.getState().participants.length === 0) {
      for (const svc of DEFAULT_PARTICIPANTS) addParticipant(svc);
    }
  }, [addParticipant]);

  const handleBroadcast = (text: string) => {
    startBroadcast(text);
    const round = useArenaStore.getState().markerRound;  // freshly incremented
    const fullText = text + buildMarkerInstruction(round);
    const done = doneMarker(round);
    for (const p of useArenaStore.getState().participants) {
      const ref = poolRef.current?.getRef(p.id);
      if (!ref) continue;
      ref.setupMarkerWatch(done);
      ref.injectAndSend(fullText);
    }
    setFlowState('awaiting_responses');
  };

  const handleDebate = () => {
    const st = useArenaStore.getState();
    const responses: Record<string, ResponseEntry> = {};
    for (const p of st.participants) {
      if (p.response) responses[p.id] = { name: p.name, text: p.response };
    }
    if (Object.keys(responses).length < 2) return;
    const nextRound = st.debateSession.rounds.length + 1;
    appendDebateRound({ roundNum: nextRound, style: 'free', guidance: '', responses });
    const round = useArenaStore.getState().markerRound;
    const done = doneMarker(round);
    for (const id of Object.keys(responses)) {
      const ref = poolRef.current?.getRef(id);
      if (!ref) continue;
      const prompt = DebateEngine.buildDebatePrompt(id, responses, 'free', nextRound, '', false);
      ref.setupMarkerWatch(done);
      ref.injectAndSend(prompt);
    }
    setFlowState('awaiting_responses');
  };

  const handleSummary = () => {
    const st = useArenaStore.getState();
    if (st.participants.length === 0) return;
    const judge = st.participants[0];
    const responses: Record<string, ResponseEntry> = {};
    for (const p of st.participants) {
      if (p.response) responses[p.id] = { name: p.name, text: p.response };
    }
    const prompt = DebateEngine.buildSummaryPrompt(
      st.debateSession.originalQuestion,
      st.debateSession.rounds,
      responses,
      '',
    );
    const round = useArenaStore.getState().markerRound + 1;
    useArenaStore.setState({ markerRound: round });
    const done = doneMarker(round);
    const ref = poolRef.current?.getRef(judge.id);
    if (ref) {
      ref.setupMarkerWatch(done);
      ref.injectAndSend(prompt);
    }
    setFlowState('summary');
  };

  const handleReset = () => resetSession();

  return (
    <SafeAreaView style={styles.container}>
      <StatusBar barStyle="dark-content" />
      <View style={styles.content}>
        <DashboardScreen
          onOpenDetail={setDetailParticipantId}
          onBroadcast={handleBroadcast}
          onDebate={handleDebate}
          onSummary={handleSummary}
          onReset={handleReset}
        />
      </View>
      <ParticipantPool ref={poolRef} activeParticipantId={detailParticipantId} />
      {detailParticipantId && (
        <DetailScreen participantId={detailParticipantId} onBack={() => setDetailParticipantId(null)} />
      )}
    </SafeAreaView>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, backgroundColor: '#f7f8fa' },
  content: { flex: 1 },
});
```
